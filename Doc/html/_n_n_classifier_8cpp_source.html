90// calculates the distance between the 2 given nodes (3d points)
91float NNClassifier::nearestNeighbour(NODE* n)
92{
93 // distance equation, typecasting to float to avoid data loss
94 return (float)sqrt(pow((this->x - n->x), 2) + pow((this->y - n->y), 2) + pow((this->z - n->z), 2));
95}

132// this function learns data based on a file given
133void NNClassifier::learnOrientation(std::string fileName)
134{
135 // declare an input file variable
136 std::ifstream fin;
137
138 // open the given file as a read file
139 fin.open(fileName);
140 if (fin.is_open())
141 {
142 // keep looping through file until reach the end of it
143 while (!fin.eof())
144 {
145 // calls the file read function in FileReadWrite.cpp
146 setAttributes(fin, *this);
147
148 // checks if there is invalid values
149 if (this->label == 7)
150 {
151 std::cout << "Invalid data found..." << std::endl;
152 }
153 else
154 {
155 // create a temp head node to traverse through and not edit original head pointer
156 NODE* temp = this->getHead();
157
158 // checks if there was an error creating the temp node
159 if (temp != nullptr)
160 {
161
162 // traverse through the node until reached tail
163 while (temp->next != nullptr)
164 {
165 temp = temp->next;
166 }
167
168 // set the node to the current data retrieved
169 temp->x = this->getX();
170 temp->y = this->getY();
171 temp->z = this->getZ();
172 temp->label = this->getLabel();
173
174 // initializing an empty node for the next loop through
175 temp->next = new NODE;
176 temp->next->x = 0;
177 temp->next->y = 0;
178 temp->next->z = 0;
179 temp->next->label = 0;
180 temp->next->next = nullptr;
181
182 }
183 }
184 }
185
186 // close file to avoid errors
187 fin.close();
188 }
189 else
190 {
191 // in case error occurs with opening file
192 std::cout << "can't open file:" << fileName << std::endl;
193 }

196// this function predicts a single point given by the user
197void NNClassifier::predictOrientation(float x, float y, float z)
198{
199 // checks if the input data is invalid or not immediately since its one value only
200 if (x < -1 || x > 1 ||
201 y < -1 || y > 1 ||
202 z < -1 || z > 1)
203 {
204 std::cout << std::endl << std::endl <<"Data entered is invalid..." << std::endl << std::endl;
205 return;
206 }
207
208 // declaring prediction variables
209 float nearestDistance;
210 int nearestLabel = 0;
211
212 // sets each attribute
213 this->setX(x);
214 this->setY(y);
215 this->setZ(z);
216
217 // create a temp head node to traverse through and not edit original head pointer
218 NODE* temp = this->head;
219
220 // calculate the distance between the current data to the first trained data
221 nearestDistance = nearestNeighbour(temp);
222
223 // traverse through the trained data till end reached
224 while (temp != nullptr)
225 {
226 // calculate the distance between the current data to all the trained data
227 if ((nearestNeighbour(temp)) < nearestDistance)
228 {
229 // if the calculated nearest distance is less than the current then set the prediction variable as that value
230 nearestDistance = nearestNeighbour(temp);
231 nearestLabel = temp->label;
232 }
233 // traverse to next node;
234 temp = temp->next;
235 }
236
237 // print out the predicted labels out to the console for the user to see
238 std::cout << std::endl << std::endl << "Predicted label:" << nearestLabel << ", " << decideOrientationName(nearestLabel) << std::endl << std::endl;
239}
